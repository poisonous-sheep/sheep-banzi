
# 欧拉路与欧拉回路 (Eulerian Path & Circuit) 🌉

## 1. 原理

求欧拉路的核心算法是 **“DFS走到黑 + 回溯时加边 + 最后倒序输出”** 。

- **图的结构拆解**：任何一条欧拉路，都可以看作是“一条基础通路”加上“挂在这条通路上的零个或多个回路”组成的 。
    
- **DFS 的运行轨迹**：
    
    1. 当你从起点开始 DFS 往下走，由于除了终点外所有点的出入度都平衡，你**一定会走到真正的终点才停下来**（此时该点没有未走过的出边了） 。
        
    2. 当 DFS 走到死胡同时，开始**回溯**。在回溯的过程中，如果发现某个途径的点还有没走过的边，DFS 会一头扎进去，把挂在这个点上的“回路”给遍历完 。
        
    3. 因为我们是在**回溯时**（也就是该点所有相邻的环都被走完时）才把点加入答案数组 `path`，所以最先加入 `path` 的其实是欧拉路的终点，最后加入的是起点 。
        
    4. 最后把 `path` 倒序，就得到了一条完美的欧拉路 。
        
- **致命的复杂度陷阱 (O(M2) vs O(M))**：
    
    - **错误做法**：在 DFS 内部用一个普通的 `for (int i = 0; i < adj[x].size(); i++)` 去找边，然后判断这条边有没有被标记过。如果一个点连了 10 万条边，你每次回溯过来都要从头扫一遍，复杂度会退化到 O(M2) 。
        
    - **正确做法 (当前弧优化/物理删边)**：走到哪条边，就把这条边彻底删掉（或移动头指针）。你的模板中使用 `pop_back()` 物理删边，完美保证了每条边只被碰一次，复杂度稳稳卡在 O(N+M) 。

C++

```
vector<int> adj[N];
int in[N], out[N];  // 记录入度和出度
vector<int> path;   // 记录最终路径
int n, m;

// 核心 DFS：找通路 + 套回路 
inline void dfs(int x) {
    // 技巧：只要还有相邻的边，就一直走下去
    // 使用 while 和 size() 配合，避免 for 循环从头遍历 
    while (adj[x].size()) {
        // 配合外部的降序排序，这里 back() 取出的是字典序最小的相邻点 
        int v = adj[x].back();
        adj[x].pop_back(); // 物理删边！防止重复遍历退化为 O(M^2) 
        dfs(v);            // 继续往下深搜
    }
    // 当该点所有连通的边都被榨干，进入死胡同回溯时，才将其加入路径 
    path.push_back(x); 
}

void Euler() {
    int st = 0, y = 0, z = 0;
    // 1. 统计起终点特征并进行合法性校验
    rep(1, n) {
        // 如果出度比入度大 1，它就是欧拉路的唯一合法起点 
        if (in[i] + 1 == out[i]) {
            st = i;
            y++; // 统计起点的个数
        }
        // 如果入度不等于出度，记录这样的点的个数 (欧拉路中只能有 2 个，或者 0 个即回路) 
        if (in[i] != out[i]) z++; 
    }
    
    // 欧拉路存在的充要条件：
    // (y == 1 && z == 2) -> 恰好 1 个起点，1 个终点 (共 2 个度不平衡的点) 
    // (z == 0) -> 所有点入度 = 出度，说明是欧拉回路 
    if (!((y == 1 && z == 2) || z == 0)) {
        cout << "No\n";
        return;
    }
    
    // 2. 寻找起点
    // 如果 z == 0 (回路)，那么 st 还是 0。此时随便找一个度数不为 0 的点作为起点即可 
    if (st == 0) {
        rep(1, n) {
            if (out[i]) {
                st = i;
                break;
            }
        }
    }
    
    // 3. 跑核心算法
    dfs(st);
    
    // 4. 将回溯记录的路径翻转，得到真实正向路径 
    reverse(path.begin(), path.end());
    
    // 5. 连通性最终校验
    // 图可能是不连通的（存在两个孤立的合法子图），如果路径点数 != 边数 + 1，说明没走完所有边 
    if (path.size() != m + 1) {
        cout << "No\n";
        return;
    }
    
    // 输出合法路径
    for (auto v : path) cout << v << " ";
}

void solve() {
    cin >> n >> m;
    rep(1, m) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        in[v]++;
        out[u]++;
    }
    
    // 核心细节：为了求“字典序最小”的欧拉路 
    // 因为 dfs 中是拿 vector 的末尾元素 (back) 来删边 
    // 所以想要优先走编号小的点，要把邻接表排成**降序 (greater)**，这样最小的元素就在最后面！ 
    rep(1, n) {
        sort(adj[i].begin(), adj[i].end(), greater<int>());
    }
    
    Euler();
}
```

无向图：
```
#include <bits/stdc++.h>
using namespace std;

// 🛠️ 现代存图结构：把“边的终点”和“边的全局唯一编号”绑在一起
struct Edge {
    int v;  // 这条边通向哪个点
    int id; // 这条边的全局唯一编号（0 到 m-1）
};

const int N = 505;  // 顶点最大编号（题目说栅栏顶点在 1~500）
const int M = 1050; // 边数最大值（题目说最多 1024 条边）

vector<Edge> adj[N]; // 邻接表存图
int deg[N];          // 记录每个点的度数（连了多少条边）
int cur[N];          // 🚀 当前弧优化数组：记录每个点遍历到了它的第几条边（充当书签）
bool vis[M];         // 🏷️ 全局边标记数组：记录某个编号的边是否已经被走过
vector<int> ans;     // 存储最终路径

// 🧠 核心 DFS 逻辑：死胡同就是终点
void dfs(int u) {
    // ⚠️ 注意这里的 int &i = cur[u]！
    // 用了引用，意味着 i++ 会真正改变 cur[u] 的值。
    // 这样下次如果再递归回到点 u，就会直接从没走过的边开始看，绝不走回头路！
    for (int &i = cur[u]; i < adj[u].size(); ) {
        // C++17 结构化绑定，直接拆出终点 v 和 边编号 id
        auto [v, id] = adj[u][i]; 
        i++; // 游标立马后移，相当于在这层 DFS 里把这条边“删掉”了
        
        // 🛑 如果这条边之前从另一个方向走过了，直接跳过
        if (vis[id]) continue; 
        
        vis[id] = true; // 标记这条边已被使用（正反两个方向同时作废）
        
        dfs(v); // 顺藤摸瓜继续往下走
    }
    // 🧱 撞南墙了：当前点 u 周围所有边都走完了
    // 把它塞入答案。因为是最先走投无路，所以它其实是整条路径的“终点”。
    ans.push_back(u); 
}

int main() {
    // ⚡ 读写提速
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int m;
    cin >> m;
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        // 🔄 无向图，所以 u->v 和 v->u 都要存。
        // 核心技巧：它们共享同一个全局编号 i！
        adj[u].push_back({v, i});
        adj[v].push_back({u, i});
        deg[u]++;
        deg[v]++;
    }

    // 🔤 字典序要求：题目要求如果有多个解，输出字典序最小的。
    // 所以在跑 DFS 之前，必须把每个点的邻居按节点编号从小到大排个序。
    for (int i = 1; i <= 500; i++) {
        sort(adj[i].begin(), adj[i].end(), [](const Edge& a, const Edge& b) {
            return a.v < b.v;
        });
    }

    int start = 0;
    
    // 🎯 第一步：尝试找奇数度数的点当起点（欧拉路径）
    // 因为要求字典序最小，从小到大遍历，找到的第一个奇数点就是我们要的起点。
    for (int i = 1; i <= 500; i++) {
        if (deg[i] % 2 != 0) {
            start = i;
            break; 
        }
    }

    // 🎯 第二步：如果全是偶数度数（欧拉回路），那就找编号最小且有连边的点当起点。
    if (start == 0) {
        for (int i = 1; i <= 500; i++) {
            if (deg[i] > 0) {
                start = i;
                break;
            }
        }
    }

    // 🚀 开始一笔画！
    if (start != 0) {
        dfs(start);
    }

    // 🔄 因为 DFS 里是最先把终点 push 进去的，所以整个数组是倒着的。
    // 输出前必须把它反转过来，才是正确的从头走到尾的顺序。
    reverse(ans.begin(), ans.end());
    
    // 🖨️ 输出答案
    for (int u : ans) {
        cout << u << "\n";
    }

    return 0;
}
```

## 3. 常见应用模型 

### A. 单词接龙 / 字符串首尾拼接 🔤

- **题意**：给定 N 个单词，要求全部连起来，前一个单词的尾字母和后一个单词的首字母相同。能不能接成一条长龙？
    
- **转化模型**：把 26 个字母当成**节点 (点集 V)**。每一个单词，看作是一条**从其首字母指向尾字母的有向边 (边集 E)** 。
    
- **原理解释**：两个相邻的单词共用了一个字母，相当于两条边首尾相连共用了一个点 。要求每个单词刚好用一次，就是在图里找一条经过所有边恰好一次的通路 —— 即**有向图的欧拉路** 。

### B. 无向图与成对建边技巧 (The `idx ^ 1` trick) 🕸️

- **应用场景**：如果是**无向图**找欧拉路，你走了一条从 u 到 v 的边，它的反向边 v 到 u 也立刻不能走了（因为同一条物理边只能走一次） 。
    
- **实现原理**：
    
    1. 用一个全局的边计数器 `cnt`，初始为 `1` 。
        
    2. 每次加边时，正向边编号 `++cnt` (比如 2)，反向边编号 `++cnt` (比如 3) 。
        
    3. 这样正反两边总是相邻的偶数和奇数。利用异或运算：`2 ^ 1 = 3`，`3 ^ 1 = 2` 。
        
    4. 开一个布尔数组 `used[M]`，每次 DFS 走到编号为 `idx` 的边，直接标记 `used[idx] = used[idx ^ 1] = true`，就能实现 O(1) 的无向边双向封锁 。
